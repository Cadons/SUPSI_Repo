df = df.sort_values("DateT").groupby("DateT").first()
df = df[["Fremont Bridge East Sidewalk","Fremont Bridge West Sidewalk"]].rename(columns={
    "Fremont Bridge East Sidewalk":"East",
    "Fremont Bridge West Sidewalk":"West",
